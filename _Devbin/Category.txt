        public const string CreateCategory = @"
                        INSERT Category 
                        ()
                        VALUES
                        ()
                        SELECT CAST(SCOPE_IDENTITY() as int)";

        public async Task<int> CreateCategoyAsync(CreateCategoryDto createCategory)
        {
            var insertId = await WithConnection<int>(
                async dbConnection =>
                {
                    var record = await dbConnection.QueryFirstOrDefaultAsync<int>(
                        CategoryQuery.CreateCategory,
                        createCategory);
                    return record;
                }
            );
            return insertId;
        }
